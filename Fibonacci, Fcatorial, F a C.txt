Funcion Fibonacci

(defun Fib (e)
  (if (< e 2)
      e
      (+ (Fib (- e 1)) (Fib (- e 2)))))
(print (Fib 8))



Funcion Factorial

(defun Fac (i)
  (if (<= i 1)
      1
      (* i (Fac (- i 1)))))
(print (Fac 5))



Funcion Farenheit a Celsius

(defun FaC (F)
  (/ (- F 32) 1.8))
(print (FaC 350))
